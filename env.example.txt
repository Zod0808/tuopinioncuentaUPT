# Configuración de OpenAI API
# Obtén tu API key en: https://platform.openai.com/api-keys
# Renombra este archivo a .env y agrega tu API key
VITE_OPENAI_API_KEY=tu_api_key_aqui

# Configuración de JSONBin.io (RECOMENDADO - Más simple que Firebase)
# 1. Ve a https://jsonbin.io/ y crea una cuenta gratuita
# 2. Crea un nuevo "Bin" (contenedor de datos)
# 3. Copia el Bin ID y tu API Key
# 4. Pega aquí:
VITE_JSONBIN_BIN_ID=tu_bin_id_aqui
VITE_JSONBIN_API_KEY=tu_api_key_aqui

# Configuración de Firebase (Alternativa a JSONBin)
# Obtén estas credenciales en: https://console.firebase.google.com/
# Crea un proyecto, ve a Configuración del proyecto > Tus aplicaciones > Web
VITE_FIREBASE_API_KEY=tu_api_key_firebase
VITE_FIREBASE_AUTH_DOMAIN=tu-proyecto.firebaseapp.com
VITE_FIREBASE_PROJECT_ID=tu-proyecto-id
VITE_FIREBASE_STORAGE_BUCKET=tu-proyecto.appspot.com
VITE_FIREBASE_MESSAGING_SENDER_ID=123456789
VITE_FIREBASE_APP_ID=1:123456789:web:abcdef
